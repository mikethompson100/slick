Slick js carousel examples

Example 1 - Basic Carousel

6 slides


    $('.carousel1').slick({});

    Using all defaults.

Example 2 - infinite and dots

6 slides


$('#carousel2').slick({
    slidesToShow: 2, 
    slidesToScroll: 2, 
    dots: true,
    infinite: false
});

Example 3 - variablewidth

6 slides


    $('#carousel3').slick({
    slidesToScroll: 1,
    variableWidth: true
    });

Example 4 - adaptiveheight (slide #2) with fade

6 slides


$('#carousel4').slick({
    adaptiveHeight: true,
    dots: true,
    fade:true,
    speed: 4000
    });

Example 5 - autoplay/autoplaySpeed

6 slides


$('#carousel5').slick({
    autoplay: true,
    autoplaySpeed: 1000
});

Example 6 - asNavFor (Carousels 6 & 7)

6 slides

Carousel 6
Carousel 7
    
$('#carousel6').slick({
  asNavFor: "#carousel7",
  dots: true,
  arrows: false
}); 
$("#carousel7").slick({
  slidesToShow: 3,
  asNavFor: "#carousel6"
});

Example 8 - asNavFor (Carousels 8 & 9 with centerMode/focusOnSelect)

6 slides

Carousel 8
Carousel 9

$('#carousel8').slick({
    asNavFor: "#carousel9",
    arrows: false,
    fade: true
}); 

$("#carousel9").slick({
    asNavFor: "#carousel8",
    focusOnSelect: true, (click on slide)
    centerMode: true,
    slidesToShow: 3   (needs to be an odd # to center)
});
